005package org.usfirst.frc330.Beachbot2013Java.commands;
006
007import edu.wpi.first.wpilibj.command.AutoSpreadsheetCommand;
008import org.usfirst.frc330.Beachbot2013Java.Robot;
009
010/**
011 *
012 * @author joe
013 */
014public class DriveEncoderGyroRel extends DriveEncoderRel{
015    double angle;
016    
017    public DriveEncoderGyroRel()
018    {
019        this(0,0,0,0,true);
020    }
021    
022    public DriveEncoderGyroRel(double distance, double angle)
023    {
024        this(distance, 0, angle, 0, true);
025    }
026    
027    public DriveEncoderGyroRel(double distance, double tolerance, double angle, double timeout, boolean stopAtEnd)
028    {
029        super(distance, tolerance, timeout, stopAtEnd);
030        this.angle = angle;
031    }
032        // Called just before this Command runs the first time
033    protected void initialize() {
034        super.initialize();
035        if (!Robot.chassis.getShiftState())
036        {
037            Robot.chassis.gyroPIDLow.setSetpoint(angle);
038            Robot.chassis.gyroPIDLow.enable();
039        }
040        else
041        {
042            Robot.chassis.gyroPIDHigh.setSetpoint(angle);
043            Robot.chassis.gyroPIDHigh.enable();
044        }
045    }
046    // Called once after isFinished returns true
047    protected void end() {
048        super.end();
049        if (stopAtEnd)
050        {
051            Robot.chassis.gyroPIDLow.disable();
052            Robot.chassis.gyroPIDHigh.disable();
053        }
054    }
055    /**
056     * The second parameter in the AutoSpreadsheet
057     * @param angle The angle to maintain while driving
058     */
059    public void setParam2(double angle) {
060        this.angle = angle;
061    }
062    
063    /**
064     * The third parameter in the AutoSpreadsheet
065     * @param tolerance The tolerance for how close to be to the distance before stopping
066     * @see edu.wpi.first.wpilibj.PIDController#setAbsoluteTolerance
067     */
068    public void setParam3(double tolerance) {
069        this.tolerance = tolerance;
070    }
071    public AutoSpreadsheetCommand copy() {
072        return new DriveEncoderGyroRel();
073    }
074}
