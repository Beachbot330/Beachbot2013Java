005package org.usfirst.frc330.Beachbot2013Java.commands;

007import com.sun.squawk.util.MathUtils;
008import edu.wpi.first.wpilibj.command.AutoSpreadsheetCommand;
009import edu.wpi.first.wpilibj.command.Command;
010import org.usfirst.frc330.Beachbot2013Java.Robot;

012 * $Log$

014/**
015 * Drive the robot to a specified waypoint (relative to where the robot started) 
016 * using the encoders and the gyro to keep straight and ramp up to full speed. Finish when
017 * one of the encoders is within the specified {@link #setParam3(double) tolerance}. 
018 * The {@link #setParam2(double) Y} location is positive to the front of the robot. 
019 * The {@link #setParam1(double) X} location is positive to the right of the robot.
020 * The robot's current angle should be close to the angle to drive to the waypoint
021 * (ie this command is not appropriate for both turning and driving).
022 * If the robot needs to turn prior to driving to the waypoint, use {@link TurnWaypoint} 
023 * first.
024 * <p>
025 * For example, to drive 10 feet forward from where the robot started, set Y to 
026 * 120 (inches) and set X to 0.
027 * A reasonable tolerance is 3 inches for normal movements. This will stop the robot
028 * when it is between 117 - 123 inches. If a smaller
029 * tolerance is used, the robot may not ever reach the tolerance, and the 
030 * {@link #setTimeout(double) timeout} may be exceeded. This will slow down the
031 * execution of future commands.
032 * 
033 * @see DriveEncoderGyroRamp
034 * @see TurnGyroWaypoint
035 * 
036 * @author Joe
037 */
038public class DriveWaypoint extends DriveEncoderGyroRampRel {
039    double x,y;
040    public DriveWaypoint(double x, double y, double tolerance, double timeout, boolean stopAtEnd) {
041        // Use requires() here to declare subsystem dependencies
042        // eg. requires(chassis);
043        super();
044        this.x = x;
045        this.y = y;
046        calcXY(x, y);
047        this.setParam3(tolerance);
048        this.setStopAtEnd(stopAtEnd);
049        this.setTimeout(timeout);
050    }
051
052    protected void initialize() {
053        calcXY(x,y);
054        super.initialize();
055    }
056    
057    
058    
059     /**
060     * The first parameter in the AutoSpreadsheet, X
061     * @param X The X component of the waypoint
062     */
063    public void setParam1(double x) {
064        this.x = x;
065    }
066    /**
067     * The second parameter in the AutoSpreadsheet, Y
068     * @param Y The Y component of the waypoint
069     */
070    public void setParam2(double y) {
071        this.y = y;
072    }
073
074    private void calcXY(double x, double y) {
075        double curX, curY, deltaX, deltaY, calcAngle, calcDistance;
076        
077        curX = Robot.chassis.getX();
078        curY = Robot.chassis.getY();
079        
080        deltaX = x - curX;
081        deltaY = y - curY;
082        
083        calcDistance = Math.sqrt(deltaX*deltaX+deltaY*deltaY);
084        calcAngle = Math.toDegrees(MathUtils.atan2(deltaX, deltaY));
085        
086        if (Math.abs(Robot.chassis.getAngle()-calcAngle)<180)
087        {
088            calcAngle=calcAngle;
089        }
090        else if (Robot.chassis.getAngle() > calcAngle)
091        {
092            calcAngle += 360;
093        }
094        else
095        {
096            calcAngle -= 360;
097        }
098        System.out.println("distance: " + calcDistance);
099        System.out.println("angle: " + calcAngle);
100        
101        super.setParam1(calcDistance);
102        super.setParam2(calcAngle);
103    }
104    
105    public Command copy() {
106        return new DriveWaypoint(0,0,0,0,false);
107    }
108
109}
