016/**
017 * Drive the robot a specified distance using encoders and the gyro to keep 
018 * straight and ramp up to full speed. Finish when
019 * one of the encoders is within the specified {@link #setParam3(double) tolerance}. 
020 * The {@link #setParam1(double) distance} is relative to the
021 * Robot's current position. The {@link #setParam2(double) angle} is relative 
022 * to the angle where the robot started. The angle should be close to the 
023 * current angle of the robot (ie this command is not appropriate for both turning and driving).
024 * If the robot needs to turn prior to driving straight, use {@link TurnGyroAbs} 
025 * first. 
026 * <p>
027 * For example, to drive 10 feet forward, set distance to 120 (inches) and set angle to 0.
028 * A reasonable tolerance is 3 inches for normal movements. This will stop the robot
029 * when it is between 117 - 123 inches. If a smaller
030 * tolerance is used, the robot may not ever reach the tolerance, and the 
031 * {@link #setTimeout(double) timeout} will be exceeded. This will slow down the
032 * execution of future commands.
033 * 
034 * @see DriveEncoderGyroRel
035 * @see DriveEncoderGyroRamp
036 * @see DriveWaypoint
037 * 
038 * @author Joe
039 */
040public class DriveEncoderGyroRampRel extends DriveEncoderGyroRamp {
041    public DriveEncoderGyroRampRel(double distance, double tolerance, double angle, double timeout, boolean stopAtEnd)
042    {
043        super(distance, tolerance, angle, timeout, stopAtEnd);
044    }
045    
046    public DriveEncoderGyroRampRel()
047    {
048        super(0,0,0,0,true);
049    }
050
051    protected void initialize() {
052        leftDistance = leftDistance + Robot.chassis.getLeftDistance();
053        rightDistance = rightDistance + Robot.chassis.getRightDistance();
054        super.initialize();
055    }
056
057    public Command copy() {
058        return new DriveEncoderGyroRampRel();
059    }
060}
