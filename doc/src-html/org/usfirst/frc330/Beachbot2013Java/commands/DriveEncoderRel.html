016/**
017 * Drive the robot a specified distance using encoders and PID only. Finish when
018 * one of the encoders is within the specified {@link #setParam2(double) tolerance}. 
019 * The {@link #setParam1(double) distance} is relative to the
020 * Robot's current position.
021 * <p>
022 * For example, to drive 10 feet forward, set distance to 120 (inches).
023 * A reasonable tolerance is 3 inches for normal movements. This will stop the robot
024 * when it is between 117 - 123 inches. If a smaller
025 * tolerance is used, the robot may not ever reach the tolerance, and the 
026 * {@link #setTimeout(double) timeout} will be exceeded. This will slow down the
027 * execution of future commands.
028 * 
029 * @see DriveEncoder
030 * @see DriveEncoderGyroRel
031 * 
032 * @author Joe
033 */
034public class DriveEncoderRel extends DriveEncoder{
035    public DriveEncoderRel(double distance, double tolerance, double timeout, boolean stopAtEnd)
036    {
037        super(distance, tolerance, timeout, stopAtEnd);
038    }
039    
040    public DriveEncoderRel()
041    {
042        super(0,0,0,true);
043    }
044
045    protected void initialize() {
046        leftDistance = leftDistance + Robot.chassis.getLeftDistance();
047        rightDistance = rightDistance + Robot.chassis.getRightDistance();
048        super.initialize();
049    }
050    
051    public Command copy() {
052        return new DriveEncoderRel();
053    }
054    
055    
056}
