039    /**
040     * Initialize SPI bus<br>
041     * Only call this method once in the program
042     *
043     * @param slot      The slot of the module the SPIBus channels are in
044     * @param clkChannel        The channel of the digital output for the clock signal.
045     * @param mosiChannel       The channel of the digital output for the written data to the slave
046     * (master-out slave-in).
047     * @param misoChannel       The channel of the digital input for the input data from the slave

050    public static void initBus(final int slot, final int clkChannel, final int mosiChannel, final int misoChannel) {
051        createdBusChannels = true;
052        initBus(new DigitalOutput(slot, clkChannel), new DigitalOutput(slot, mosiChannel), new DigitalInput(slot, misoChannel));
053    }

055    /**
056     * Initialize SPI bus<br>
057     * Only call this method once in the program
058     *
059     * @param clk       The digital output for the clock signal.
060     * @param mosi      The digital output for the written data to the slave
061     * (master-out slave-in).
062     * @param miso      The digital input for the input data from the slave
063     * (master-in slave-out).
064     */
065    public static void initBus(final DigitalOutput clk, final DigitalOutput mosi, final DigitalInput miso) {
066        if (spi == null)
067            spi = new tSPI();
068        else
069            throw new SPIDevice.BadSPIConfigException("The SPI bus can only be initialized once");
070        
071        clkChannel = clk;
072        mosiChannel = mosi;
073        misoChannel = miso;
074
075
076        tSPI.writeChannels_SCLK_Module(clk.getModuleForRouting());
077        tSPI.writeChannels_SCLK_Channel(clk.getChannelForRouting());
078
079
080        if (mosi != null) {
081            tSPI.writeChannels_MOSI_Module(mosi.getModuleForRouting());
082            tSPI.writeChannels_MOSI_Channel(mosi.getChannelForRouting());
083        } else {
084            tSPI.writeChannels_MOSI_Module(0);
085            tSPI.writeChannels_MOSI_Channel(0);
086        }
087
088        if (miso != null) {
089            tSPI.writeChannels_MISO_Module(miso.getModuleForRouting());
090            tSPI.writeChannels_MISO_Channel(miso.getChannelForRouting());
091            tSPI.writeConfig_WriteOnly(false);//TODO check these are right
092        } else {
093            tSPI.writeChannels_MISO_Module(0);
094            tSPI.writeChannels_MISO_Channel(0);
095            tSPI.writeConfig_WriteOnly(true);
096        }
097
098
099        tSPI.writeChannels_SS_Module(0);
100        tSPI.writeChannels_SS_Channel(0);
101
102        tSPI.strobeReset();
103        tSPI.strobeClearReceivedData();
104    }
105
106    /**
107     * Cleanup the SPI Bus
108     */
109    public static void freeBus() {
110        if (spi != null) {
111            spi.Release();
112            spi = null;
113        }
114        if(createdBusChannels){
115            clkChannel.free();
116            mosiChannel.free();
117            misoChannel.free();
118        }
119        createdBusChannels = false;
120    }

122    /**
123     * Perform a SPI transfer with the length of the selected device's current
124     * configuration
125     *
126     * @param value     The value to write to the bus
127     */
128    private static long transferStatic(long value) {
129        synchronized (semaphore) {
130            tSPI.strobeClearReceivedData();
131
132            tSPI.writeDataToLoad(value);
133            tSPI.strobeLoad();
134
135            if (!tSPI.readConfig_WriteOnly())
136            {
137                while (tSPI.readReceivedElements() == 0) {
138                    Thread.yield();
139                }
140
141                tSPI.strobeReadReceivedData();
142                return tSPI.readReceivedData();
143            }
144            else
145                return(0);
146        }
147    }

315    /**
316     * Perform a SPI transfer with the length of this device's current
317     * configuration. This will select the device, transfer the data and then
318     * deselect the device
319     *
320     * @param writeValue        The value to write to the device
321     * @param numBits   The number of bits to write/read
322     */
323    public long transfer(long writeValue, int numBits) {
324        long[] readValue;
325        synchronized (semaphore) {
326            readValue = transfer(new long[]{writeValue}, new int[]{numBits});
327        }
328        return readValue[0];
329    }
330
331    /**
332     * Perform a SPI transfer where an array of bits are written and read. The
333     * number of bits to write and read is specified in numBits<br> The whole
334     * transfer will occur with the cs line held active throughout
335     *
336     * @param writeValues       The value to write to the device
337     * @param numBits   The number of bits to write/read
338     */
339    public long[] transfer(long[] writeValues, int[] numBits) {
340        if (writeValues.length != numBits.length) {
341            throw new SPIDevice.BadSPIConfigException("The number of values to write does not match array of data lengths");
342        }
343        for (int i = 0; i < numBits.length; ++i) {
344            if (numBits[i] < 1 || numBits[i] > 32) {
345                throw new SPIDevice.BadSPIConfigException("All values in the data length must be >0 and <=32");
346            }
347        }
348        long[] readValues = new long[writeValues.length];
349        synchronized (semaphore) {
350            tSPI.writeConfig_MSBfirst(bitOrder);
351            tSPI.writeConfig_ClockHalfPeriodDelay(clockHalfPeriodDelay);
352            tSPI.writeConfig_ClockPolarity(clockPolarity);
353            tSPI.writeConfig_DataOnFalling(dataOnTrailing);
354
355            
356            //cs.set(csActiveHigh);
357            tSPI.writeConfig_FramePolarity(!csActiveHigh);
358            //Set up FPGA for chip select
359//            tSPI.writeConfig_LatchLast(false);
360//            tSPI.writeConfig_LatchFirst(false);
361            writeFrameMode(frameMode);
362            tSPI.writeChannels_SS_Module(cs.getModuleForRouting());
363            tSPI.writeChannels_SS_Channel(cs.getChannelForRouting());
364
365            for (int i = 0; i < writeValues.length; ++i) {
366                tSPI.writeConfig_BusBitWidth(numBits[i]);
367                readValues[i] = transferStatic(writeValues[i]);
368            }
369
370            //cs.set(!csActiveHigh);
371            //JROSS these two lines cause a short chip select on non-default clock rates
372            //(possibly only in write only mode
373            //commenting these out might break having multiple devices on the bus.
374//            tSPI.writeChannels_SS_Module(0);  
375//            tSPI.writeChannels_SS_Channel(0);
376        }
377        return readValues;
378    }

438    /**
439     * Set the frequence of the clock when sending data
440     *
441     * @param hz        the frequence of the clock in hz
442     *
443     * @see #MIN_CLOCK_FREQUENCY
444     * @see #MAX_CLOCK_FREQUENCY
445     */
446    public final void setClockRate(double hz) {
447        int delay = 0;
448        // TODO: compute the appropriate values based on digital loop timing
449        if (hz <= 76628.4) {
450            double v = (1.0 / hz) / 1.305e-5;
451            int intv = (int) v;
452            if (v - intv > 0.5) {
453                delay = intv;
454            } else {
455                delay = intv - 1;
456            }
457        } else {
458            throw new SPIDevice.BadSPIConfigException("Clock Rate too high. Hz: " + hz);
459        }
460        if (delay > 255) {
461            throw new SPIDevice.BadSPIConfigException("Clock Rate too low. Hz: " + hz);
462        }
463
464        clockHalfPeriodDelay = delay;
465    }
466    private void writeFrameMode(char frameMode)
467    {
468        switch (frameMode)
469        {
470            case 0: default:
471                tSPI.writeConfig_LatchFirst(false);
472                tSPI.writeConfig_LatchLast(false);
473                break;
474            case 1:
475                tSPI.writeConfig_LatchFirst(true);
476                tSPI.writeConfig_LatchLast(false);
477                break;
478            case 2:
479                tSPI.writeConfig_LatchFirst(false);
480                tSPI.writeConfig_LatchLast(true);
481                break;
482            case 3:
483                tSPI.writeConfig_LatchFirst(true);
484                tSPI.writeConfig_LatchLast(true);
485                break;                
486        }
487    }
